25// Calculate padded length according to NIP-44 spec
26size_t calcPaddedLen(size_t unpadded_len) {
27 logInfo("calcPaddedLen input: " + String(unpadded_len));
28
29 if (unpadded_len < 1 || unpadded_len > 65535) {
30 logInfo("calcPaddedLen failed: Invalid length " + String(unpadded_len));
31 return 0;
32 }
33
34 // For messages <= 32 bytes, pad to 32
35 if (unpadded_len <= 32) {
36 logInfo("calcPaddedLen: using 32 byte padding");
37 return 32;
38 }
39
40 // Calculate next power of 2 based on (len-1)
41 size_t len_minus_one = unpadded_len - 1;
42 size_t next_power = 1;
43
44 while (next_power <= len_minus_one) {
45 next_power <<= 1;
46 logInfo("next_power now: " + String(next_power));
47 }
48
49 // Calculate chunk size
50 size_t chunk = next_power <= 256 ? 32 : next_power / 8;
51 logInfo("chunk size: " + String(chunk));
52
53 // Calculate final padded length
54 return chunk * (((unpadded_len - 1) / chunk) + 1);
55}

131void hkdf_sha256(const uint8_t *salt, size_t salt_len,
132 const uint8_t *ikm, size_t ikm_len,
133 const uint8_t *info, size_t info_len,
134 uint8_t *okm, size_t okm_len) {
135 // HKDF-Extract phase
136 uint8_t prk[32];
137 mbedtls_md_context_t ctx;
138 mbedtls_md_init(&ctx);
139 mbedtls_md_setup(&ctx, mbedtls_md_info_from_type(MBEDTLS_MD_SHA256), 1);
140
141 // Extract phase
142 mbedtls_md_hmac_starts(&ctx, salt, salt_len);
143 mbedtls_md_hmac_update(&ctx, ikm, ikm_len);
144 mbedtls_md_hmac_finish(&ctx, prk);
145
146 // Expand phase
147 uint8_t T[32] = {0}; // T(0) is empty string
148 uint8_t counter = 1;
149 size_t offset = 0;
150
151 while (okm_len > 0) {
152 mbedtls_md_hmac_starts(&ctx, prk, 32);
153
154 if (offset > 0) {
155 mbedtls_md_hmac_update(&ctx, T, 32);
156 }
157
158 if (info_len > 0) {
159 mbedtls_md_hmac_update(&ctx, info, info_len);
160 }
161
162 mbedtls_md_hmac_update(&ctx, &counter, 1);
163 mbedtls_md_hmac_finish(&ctx, T);
164
165 size_t todo = (okm_len < 32) ? okm_len : 32;
166 memcpy(okm + offset, T, todo);
167
168 counter++;
169 offset += todo;
170 okm_len -= todo;
171 }
172
173 mbedtls_md_free(&ctx);
174}
