71bool AXS15231B_Touch::update() {
72 // Check if interrupt occured, if there was an interrupt get data from touch controller and clear flag
73 if (!touch_int) {
74 return false;
75 } else {
76 touch_int = false;
77 }
78
79 uint8_t tmp_buf[8] = {0};
80 // Command to read data from controller
81 // This commands configures the controller to read only the position of the first finger
82 static const uint8_t read_touchpad_cmd[8] = {0xB5, 0xAB, 0xA5, 0x5A, 0x00, 0x00, 0x00, 0x08};
83
84 // Send command to controller
85 Wire.beginTransmission(addr);
86 Wire.write(read_touchpad_cmd, sizeof(read_touchpad_cmd));
87 Wire.endTransmission();
88
89 // Read response from controller
90 Wire.requestFrom(addr, sizeof(tmp_buf));
91 for(int i = 0; i < sizeof(tmp_buf) && Wire.available(); i++) {
92 tmp_buf[i] = Wire.read();
93 }
94
95 // Extract X and Y coordinates from response
96 uint16_t raw_X = AXS_GET_POINT_X(tmp_buf);
97 uint16_t raw_Y = AXS_GET_POINT_Y(tmp_buf);
98
99 // Validate data
100 if (point_X || point_Y) {
101 if (raw_X > x_real_max) raw_X = x_real_max;
102 if (raw_X < x_real_min) raw_X = x_real_min;
103 if (raw_Y > y_real_max) raw_Y = y_real_max;
104 if (raw_Y < y_real_min) raw_Y = y_real_min;
105 }
106
107 // Correct offset if enabled
108 uint16_t x_max, y_max;
109 if (en_offset_correction) {
110 correctOffset(&raw_X, &raw_Y);
111 x_max = x_ideal_max;
112 y_max = y_ideal_max;
113 } else {
114 x_max = x_real_max;
115 y_max = y_real_max;
116 }
117
118 // Align X and Y according to rotation
119 switch (rotation) {
120 case 0:
121 point_X = raw_X;
122 point_Y = raw_Y;
123 break;
124 case 1:
125 point_X = raw_Y;
126 point_Y = x_max - raw_X;
127 break;
128 case 2:
129 point_X = x_max - raw_X;
130 point_Y = y_max - raw_Y;
131 break;
132 case 3:
133 point_X = y_max - raw_Y;
134 point_Y = raw_X;
135 break;
136 default:
137 break;
138 }
139
140 return true;
141}
