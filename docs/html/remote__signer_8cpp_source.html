677 void processLoop() {
678 if (!signer_initialized) {
679 return;
680 }
681
682 // Update time
683 timeClient.update();
684 unixTimestamp = timeClient.getEpochTime();
685
686 // Process WebSocket events
687 webSocket.loop();
688
689 // Send periodic ping
690 unsigned long now = millis();
691 if (now - last_ws_ping > Config::WS_PING_INTERVAL) {
692 sendPing();
693 last_ws_ping = now;
694 }
695
696 // Periodic status updates every 5 seconds
697 static unsigned long last_status_update = 0;
698 if (now - last_status_update > 5000) {
699 displayConnectionStatus(isConnected());
700 last_status_update = now;
701 }
702
703 // Debug: Log connection health every 30 seconds
704 static unsigned long last_debug_log = 0;
705 if (now - last_debug_log > 30000) {
706 if (isConnected()) {
707 Serial.println("RemoteSigner::processLoop() - Connection healthy. Last message: " + String((now - last_ws_message_received) / 1000) + "s ago");
708 } else {
709 Serial.println("RemoteSigner::processLoop() - Not connected. Manual reconnect needed: " + String(manual_reconnect_needed ? "Yes" : "No"));
710 }
711 last_debug_log = now;
712 }
713
714 // Check connection health
715 if (isConnected() && (now - last_ws_message_received > Config::CONNECTION_TIMEOUT)) {
716 Serial.println("RemoteSigner::processLoop() - Connection timeout detected");
717 Serial.println("Last message received: " + String((now - last_ws_message_received) / 1000) + "s ago");
718 disconnect();
719 manual_reconnect_needed = true;
720 }
721
722 // Handle manual reconnection with exponential backoff
723 if (manual_reconnect_needed && !connection_in_progress && !isConnected()) {
724 unsigned long backoff_delay = Config::MIN_RECONNECT_INTERVAL * (1 << min(reconnection_attempts, 5)); // Cap at 32x base interval
725

727 if (reconnection_attempts < Config::MAX_RECONNECT_ATTEMPTS) {
728 Serial.println("RemoteSigner::processLoop() - Attempting manual reconnection #" + String(reconnection_attempts + 1));
729 Serial.println("Backoff delay was: " + String(backoff_delay) + "ms");
730
731 connectToRelay();
732 last_reconnect_attempt = now;
733 reconnection_attempts++;
734 } else {
735 Serial.println("RemoteSigner::processLoop() - Max reconnection attempts reached, giving up");
736 // reboot the device
737 ESP.restart();
738 manual_reconnect_needed = false;
739 reconnection_attempts = 0;
740
741 // Update status display immediately
742 displayConnectionStatus(false);
743
744 if (status_callback) {
745 status_callback(false, "Connection failed permanently");
746 }
747 }
748 }
749 }
