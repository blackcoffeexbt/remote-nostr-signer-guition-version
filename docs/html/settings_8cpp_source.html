487 if (code == LV_EVENT_CLICKED) {
488 const char* current_pin_entered = lv_textarea_get_text(pin_current_textarea);
489 const char* new_pin = lv_textarea_get_text(pin_new_textarea);
490 const char* verify_pin = lv_textarea_get_text(pin_verify_textarea);
491
492 if (!verifyPin(String(current_pin_entered))) {
493 UI::showMessage("Error", "Current PIN is incorrect");
494 return;
495 }
496
497 if (String(new_pin) != String(verify_pin)) {
498 UI::showMessage("Error", "New PINs do not match");
499 return;
500 }
501
502 if (strlen(new_pin) < 4) {
503 UI::showMessage("Error", "PIN must be at least 4 digits");
504 return;
505 }
506
507 setCurrentPin(String(new_pin));
508 UI::showMessage("Success", "PIN updated successfully");
509 hidePinManagementScreen();
510 }
