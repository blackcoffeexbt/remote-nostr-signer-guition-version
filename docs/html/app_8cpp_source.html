593 void initSleepMode()
594 {
595 // Configure GPIO 12 as input with pulldown for wake button
596 gpio_config_t io_conf;
597 io_conf.intr_type = GPIO_INTR_DISABLE;
598 io_conf.mode = GPIO_MODE_INPUT;
599 io_conf.pin_bit_mask = (1ULL << Config::WAKE_PIN);
600 io_conf.pull_down_en = GPIO_PULLDOWN_ENABLE;
601 io_conf.pull_up_en = GPIO_PULLUP_DISABLE;
602 gpio_config(&io_conf);
603
604 // Initialize activity timer
605 resetActivityTimer();
606
607 // Start sleep monitoring timer if enabled
608 if (sleep_enabled)
609 {
610 sleep_check_timer = lv_timer_create(sleepCheckCallback, Config::SLEEP_CHECK_INTERVAL, NULL);
611 }
612
613 // Check if we're waking up from deep sleep
614 esp_sleep_wakeup_cause_t wakeup_reason = esp_sleep_get_wakeup_cause();
615 if (wakeup_reason != ESP_SLEEP_WAKEUP_UNDEFINED)
616 {
617 exitSleepMode(); // Handle wake up
618 }
619
620 Serial.println("Sleep management initialized");
621 Serial.println("Wake pin: GPIO " + String(Config::WAKE_PIN));
622 Serial.println("Light sleep timeout: " + String(LIGHT_SLEEP_TIMEOUT / 1000) + " seconds");
623 Serial.println("Deep sleep timeout: " + String(DEEP_SLEEP_TIMEOUT / 1000) + " seconds");
624 }

649 void sleepCheckCallback(lv_timer_t *timer)
650 {
651 if (!sleep_enabled)
652 {
653 return;
654 }
655
656 unsigned long current_time = millis();
657 unsigned long inactive_time = current_time - last_activity_time;
658
659 // Don't sleep if invoice processing is active
660 if (UI::isInvoiceProcessing())
661 {
662 return;
663 }
664
665 // Don't sleep if WiFi is not connected (may be in setup mode)
666 if (!WiFiManager::isConnected())
667 {
668 return;
669 }
670
671 // Only allow sleep from the keypad screen
672 if (UI::getCurrentScreen() != UI::SCREEN_KEYPAD)
673 {
674 return;
675 }
676
677 // Check for deep sleep timeout (2 minutes)
678 if (inactive_time >= DEEP_SLEEP_TIMEOUT)
679 {
680 Serial.println("Deep sleep timeout reached (" + String(inactive_time / 1000) + "s), entering deep sleep");
681 enterSleepMode();
682 }
683 // Check for light sleep timeout (10 seconds) - only if not already in light sleep
684 else if (!in_light_sleep && inactive_time >= LIGHT_SLEEP_TIMEOUT)
685 {
686 Serial.println("Light sleep timeout reached (" + String(inactive_time / 1000) + "s), entering light sleep");
687 enterLightSleepMode();
688 }
689 }
